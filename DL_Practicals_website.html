import tensorflow as tf

def get_matrix_input(rows, cols, name):
    print(f"\nEnter elements for {name} matrix:")
    matrix = []
    for i in range(rows):
        row = list(map(float, input(f"Enter row {i + 1} (space separated): ").split()))
        if len(row) != cols:
            raise ValueError("Number of elements does not match the given columns")
        matrix.append(row)
    return matrix

rows_a = int(input("Enter number of rows for Matrix A: "))
cols_a = int(input("Enter number of columns for Matrix A: "))
rows_b = int(input("Enter number of rows for Matrix B: "))
cols_b = int(input("Enter number of columns for Matrix B: "))

if cols_a != rows_b:
    print("\nMatrix multiplication not possible!")
    print("Columns of Matrix A must equal rows of Matrix B")
else:
    matrix_a = get_matrix_input(rows_a, cols_a, "Matrix A")
    matrix_b = get_matrix_input(rows_b, cols_b, "Matrix B")

    tensor_a = tf.constant(matrix_a, dtype=tf.float32)
    tensor_b = tf.constant(matrix_b, dtype=tf.float32)

    print("\nMatrix A:")
    print(tensor_a.numpy())
    print("\nMatrix B:")
    print(tensor_b.numpy())

    result = tf.matmul(tensor_a, tensor_b)

    print("\nMatrix Multiplication Result:")
    print(result.numpy())

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
import seaborn as sns
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import confusion_matrix, classification_report
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense

dataset = load_breast_cancer()
features = dataset.data
labels = dataset.target

print("Target classes:", dataset.target_names)
print("Unique labels:", np.unique(labels))

features_train, features_test, labels_train, labels_test = train_test_split(
    features, labels, test_size=0.2, random_state=42

features_test = scaler.transform(features_test)

model = Sequential([

    Dense(8, activation='relu'),
    Dense(1, activation='sigmoid')
])

model.compile(
    optimizer='adam',
    loss='binary_crossentropy',
    metrics=['accuracy']
)

training_history = model.fit(
    features_train, labels_train,
    epochs=30,
    batch_size=32,
    validation_split=0.2
)

plt.plot(training_history.history['accuracy'], label='Train')
plt.plot(training_history.history['val_accuracy'], label='Validation')
plt.xlabel('Epochs')
plt.ylabel('Accuracy')
plt.title('Accuracy Curve')
plt.legend()
plt.show()

plt.plot(training_history.history['loss'], label='Train')
plt.plot(training_history.history['val_loss'], label='Validation')
plt.xlabel('Epochs')
plt.ylabel('Loss')
plt.title('Loss Curve')
plt.legend()
plt.show()

test_loss, test_accuracy = model.evaluate(features_test, labels_test)
print("Test Accuracy:", test_accuracy)

predicted_labels = (model.predict(features_test) > 0.5).astype("int32")

conf_matrix = confusion_matrix(labels_test, predicted_labels)
sns.heatmap(conf_matrix, annot=True, fmt='d',
            xticklabels=['Malignant', 'Benign'],
            yticklabels=['Malignant', 'Benign'])

Aim: Implementing regularization to avoid overfitting in binary classification code

Prerequisites
Libraries: tensorflow, numpy, matplotlib, scikit-learn
Dataset: Breast Cancer Wisconsin — built-in with scikit-learn, no download needed
pip install tensorflow numpy matplotlib scikit-learn
Python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout
from tensorflow.keras.regularizers import l2

dataset = load_breast_cancer()
features = dataset.data
labels = dataset.target

features_train, features_test, labels_train, labels_test = train_test_split(
    features, labels, test_size=0.2, random_state=42
)

scaler = StandardScaler()
features_train = scaler.fit_transform(features_train)
features_test = scaler.transform(features_test)

model_no_reg = Sequential([
    Dense(64, activation='relu', input_shape=(features_train.shape[1],)),
    Dense(32, activation='relu'),
    Dense(1, activation='sigmoid')
])

model_no_reg.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
history_no_reg = model_no_reg.fit(
    features_train, labels_train, epochs=50, validation_split=0.2, verbose=0
)

model_with_reg = Sequential([
    Dense(64, activation='relu', kernel_regularizer=l2(0.01),
          input_shape=(features_train.shape[1],)),
    Dropout(0.5),
    Dense(32, activation='relu', kernel_regularizer=l2(0.01)),
    Dropout(0.5),
    Dense(1, activation='sigmoid')
])

model_with_reg.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
history_with_reg = model_with_reg.fit(
    features_train, labels_train, epochs=50, validation_split=0.2, verbose=0
)

plt.figure(figsize=(8, 5))
plt.plot(history_no_reg.history['val_accuracy'], label='Without Regularization')
plt.plot(history_with_reg.history['val_accuracy'], label='With L2 + Dropout')
plt.title("Validation Accuracy Comparison")
plt.xlabel("Epochs")
plt.ylabel("Accuracy")
plt.legend()
plt.show()

loss_no_reg, accuracy_no_reg = model_no_reg.evaluate(features_test, labels_test, verbose=0)
loss_with_reg, accuracy_with_reg = model_with_reg.evaluate(features_test, labels_test, verbose=0)

print("Test Accuracy Without Regularization:", accuracy_no_reg)
print("Test Accuracy With L2 + Dropout:", accuracy_with_reg)

Aim: Evaluating Feed Forward Deep Network for Multiclass Classification using KFold Cross-Validation

Prerequisites

Dataset: Iris — built-in with scikit-learn, no download needed
pip install tensorflow numpy matplotlib seaborn scikit-learn

import numpy as np
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers
from sklearn.datasets import load_iris
from sklearn.model_selection import KFold
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score, confusion_matrix
from tensorflow.keras.utils import to_categorical
import seaborn as sns
import matplotlib.pyplot as plt

print("Loading Iris Dataset...\n")
iris_data = load_iris()
features = iris_data.data
labels = iris_data.target

labels_encoded = to_categorical(labels)
print("Dataset Shape:", features.shape)

scaler = StandardScaler()
features_scaled = scaler.fit_transform(features)

num_folds = 5
kfold = KFold(n_splits=num_folds, shuffle=True, random_state=42)
fold_number = 1
fold_accuracies = []

for train_indices, test_indices in kfold.split(features_scaled):
    print(f"\nTraining for Fold {fold_number}...\n")

    features_train = features_scaled[train_indices]
    features_test = features_scaled[test_indices]
    labels_train = labels_encoded[train_indices]
    labels_test = labels_encoded[test_indices]

    model = keras.Sequential([
        layers.Dense(64, activation='relu', input_shape=(features.shape[1],)),
        layers.Dense(32, activation='relu'),
        layers.Dense(3, activation='softmax')
    ])

    model.compile(
        optimizer='adam',
        loss='categorical_crossentropy',
        metrics=['accuracy']
    )

    model.fit(features_train, labels_train, epochs=50, batch_size=16, verbose=0)

    predicted_probabilities = model.predict(features_test)
    predicted_classes = np.argmax(predicted_probabilities, axis=1)
    true_classes = np.argmax(labels_test, axis=1)

    fold_accuracy = accuracy_score(true_classes, predicted_classes)
    print(f"Accuracy for Fold {fold_number}: {fold_accuracy:.4f}")

    fold_accuracies.append(fold_accuracy)
    fold_number += 1

print("\nK-Fold Cross Validation Results:")
for i in range(len(fold_accuracies)):
    print(f"Fold {i + 1} Accuracy: {fold_accuracies[i]:.4f}")

print("\nAverage Accuracy:", np.mean(fold_accuracies))

conf_matrix = confusion_matrix(true_classes, predicted_classes)
plt.figure()
sns.heatmap(conf_matrix, annot=True, fmt='d')
plt.xlabel("Predicted")
plt.ylabel("Actual")
plt.title("Confusion Matrix")
plt.show()

    batch_size=64,
    validation_data=(test_images, test_labels)

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers
from sklearn.preprocessing import MinMaxScaler
import yfinance as yf

stock_data = yf.download("AAPL", start="2015-01-01", end="2024-01-01")
close_prices = stock_data[['Close']]

print("Dataset shape:", close_prices.shape)
print(close_prices.head())

price_values = close_prices.values
scaler = MinMaxScaler(feature_range=(0, 1))
scaled_prices = scaler.fit_transform(price_values)

def create_sequences(data, window_size):
    sequence_features, sequence_targets = [], []
    for i in range(len(data) - window_size):
        sequence_features.append(data[i:i + window_size])
        sequence_targets.append(data[i + window_size])
    return np.array(sequence_features), np.array(sequence_targets)

lookback_window = 30
sequence_x, sequence_y = create_sequences(scaled_prices, lookback_window)

split_index = int(0.8 * len(sequence_x))
train_x, test_x = sequence_x[:split_index], sequence_x[split_index:]
train_y, test_y = sequence_y[:split_index], sequence_y[split_index:]

model = keras.Sequential([
    layers.LSTM(50, return_sequences=True, input_shape=(lookback_window, 1)),
    layers.LSTM(50),
    layers.Dense(1)
])

model.compile(optimizer='adam', loss='mean_squared_error')
model.summary()

training_history = model.fit(
    train_x, train_y,
    epochs=20,
    batch_size=32,
    validation_data=(test_x, test_y)
)

predicted_scaled = model.predict(test_x)

predicted_prices = scaler.inverse_transform(predicted_scaled)
actual_prices = scaler.inverse_transform(test_y)

plt.figure(figsize=(12, 6))
plt.plot(actual_prices, label="Actual Price")
plt.plot(predicted_prices, label="Predicted Price")
plt.title("AAPL Stock Price Prediction using LSTM")
plt.xlabel("Time")
plt.ylabel("Stock Price")
plt.legend()
plt.show()
